    Initialize from a Decoder. The decoder uses string value keywords for all the Int base enums defined by this struct where the keywords listed in the table below.

    Property Keywords
    keyboardType [ default, asciiCapable, numbersAndPunctuation, URL, 
                               "numberPad", "phonePad", "namePhonePad", "emailAddress",
                           "decimalPad", "twitter", "webSearch", "asciiCapableNumberPad"]   |

    | autocapitalizationType | [none, words, sentences, allCharacters] | | spellCheckingType | [default, no, yes] | | autocorrectionType | [default, no, yes] |

           let json = """
           {
           "textValidator" : "[A:C]",

           "maximumLength" : 10,
           "autocapitalizationType" : "words",
           "keyboardType" : "asciiCapable",
           "isSecureTextEntry" : true
           }
       """.data(using: .utf8)! // our data in native (JSON) format
